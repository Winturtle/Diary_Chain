using namespace std;

void maximum(int x, int y, int z) {
	int temp;
	if (x >= y) {
		temp = x;
	}
	else { 
		temp = y; 
	}
	if (temp >= z) {
		cout << "maximum= " << temp << "\n";
			}
	else {
		cout << "maximum = " << z << "\n"; 
	}
}

int main()
{
	maximum(48, 65, 31);
}

#include <iostream>
using namespace std;

int main()
{
	int x, a, b;
	float c;
	double d;

	a = 10;
	b = 20;
	c = 45.55;
	d = 472.1578;

	x = (a - b) + (c * d) / b;
	cout << "result= " << (a - b) + (c * d) / b << "\n";
	cout << "x= " << x << "\n";
}
